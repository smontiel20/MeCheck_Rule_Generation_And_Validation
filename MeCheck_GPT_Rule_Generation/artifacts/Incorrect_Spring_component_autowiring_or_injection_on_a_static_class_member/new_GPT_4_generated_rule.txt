Rule incorrect_static_member_autowiring {
  for (class c in getClasses()) {
    if (hasAnnotation(c, "Component")) {
      for (field f in getFields(c)) {
        if (hasAnnotation(f, "Autowired") OR hasAnnotation(f, "Inject")) {
          assert (NOT isStatic(f)) {
            msg("Field: %s in class: %s is a static member and should not be autowired or injected", getName(f), getFQN(c));
          }
        }
      }
      for (method m in getMethods(c)) {
        if (hasAnnotation(m, "Autowired") OR hasAnnotation(m, "Inject")) {
          assert (NOT isStatic(m)) {
            msg("Method: %s in class: %s is a static member and should not be autowired or injected", getName(m), getFQN(c));
          }
        }
      }
    }
  }
}
