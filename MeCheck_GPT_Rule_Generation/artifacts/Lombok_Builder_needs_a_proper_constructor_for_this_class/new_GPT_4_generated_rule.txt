Rule lombok-builder-all-args-constructor {
  for (class c in getClasses()) {
    if (hasAnnotation(c, '@lombok.Builder')) {
      assert ( exists(method m in getConstructors(c)) (getArg(m) == getType(c)) ) {
        msg("Class: %s annotated with @lombok.Builder does not have a proper all-args constructor", getFQN(c));
      }
    }
  }
}
