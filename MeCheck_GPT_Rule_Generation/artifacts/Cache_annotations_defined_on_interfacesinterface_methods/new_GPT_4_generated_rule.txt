Rule cache_annotations_on_interfaces {
  for (class c in getClasses()) {
    if (isInterface(c)) {
      if (hasAnnotation(c, '@Cache*')) {
        msg('Interface: %s should not be annotated with @Cache*', getName(c));
      }
      for (method m in getMethods(c)) {
        if (hasAnnotation(m, '@Cache*')) {
          msg('Method: %s in interface: %s should not be annotated with @Cache*', getName(m), getName(c));
        }
      }
    }
  }
}
